def gradient(X, y, weights):
    y_pred = np.dot(X, weights)
    gradient = (2/X.shape[0]) * np.dot(X.T, (y_pred - y))
    return gradient

weights_RMSP = np.linalg.inv(X.T @ X) @ X.T @ y

# Set learning rate, decay rate, and number of iterations
learning_rate = 0.01
B = 0.7
num_iterations = 100000

# Initialize sum of squared gradients
S = np.zeros(weights_RMSP.shape)

# Perform RMSprop optimization
for i in tqdm(range(num_iterations)):
    
    # Calculate gradient of mean squared error with respect to weights_RMSP
    grad = gradient(X, y, weights_RMSP)
    
    # Calculate sum of squared gradients using decay rate
    S = B * S + (1 - B) * grad ** 2
    
    # Update weights_RMSP using RMSprop formula
    weights_RMSP -= grad * (learning_rate / np.sqrt(S))

weights_SGD = np.linalg.inv(X.T @ X) @ X.T @ y

# Set learning rate
learning_rate = 0.00000001
for i in tqdm(range(len(X))):
    # Choose a random sample from the data

    x = X[i]
    y_true = y[i]

    # Calculate predicted value using current weights
    y_pred = np.dot(x, weights_SGD)

    # Calculate gradient of mean squared error with respect to weights
    grad = 2 * (y_pred - y_true) * x

    # Update weights using stochastic gradient descent formula
    weights_SGD -= learning_rate * grad
